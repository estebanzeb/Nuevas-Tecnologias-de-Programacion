CREATE TABLE customer(
	id int unsigned auto_increment primary key,
	name varchar(25) NOT NULL,
    status varchar(25) NOT NULL,
    mobile varchar(25) NOT NULL
);

CREATE TABLE invoice(
	id_invoice int unsigned auto_increment primary key,
	number varchar(25) NOT NULL,
    date date NOT NULL,
    id_user int unsigned,
    foreign key(id_user) references customer(id),
    price varchar(25) NOT NULL,
    balance varchar(25) NOT NULL
);

SELECT id, date, id_user, price, balance from customer as us 
INNER JOIN invoice as I on I.id_user= us.id
ORDER BY us.id ASC
